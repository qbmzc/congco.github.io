## for循环

```java
package com.taimeitech.paas.fs.web.service.impl;

import org.apache.commons.lang3.StringUtils;

import java.util.concurrent.TimeUnit;

/**
 * @author cong
 * @date 2021/9/22 17:04
 */
public class Test {
    public static void main(String[] args) {
        String s = "";
        for (int i = 0; i < 3; i++) {
            if (StringUtils.isBlank(s)) {
                if (i == 2)
                    s = "ok";
            } else {
                try {
                    TimeUnit.SECONDS.sleep(3);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        }
        System.out.println(s);
    }
}
```

```java
package com.cong.officelotus.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.retry.annotation.EnableRetry;
import org.springframework.retry.backoff.FixedBackOffPolicy;
import org.springframework.retry.policy.SimpleRetryPolicy;
import org.springframework.retry.support.RetryTemplate;

/**
 * @author cong
 * @date 2021/9/18 18:12
 */
@Configuration
@EnableRetry
public class RetryTemplateConfig {

    @Bean
    public RetryTemplate retryTemplate() {
        SimpleRetryPolicy retryPolicy = new SimpleRetryPolicy();
        retryPolicy.setMaxAttempts(4);
        FixedBackOffPolicy backOffPolicy = new FixedBackOffPolicy();
        backOffPolicy.setBackOffPeriod(3000);
        RetryTemplate template = new RetryTemplate();
        template.setRetryPolicy(retryPolicy);
        template.setBackOffPolicy(backOffPolicy);
        return template;
    }
}
```

```java
package com.cong.officelotus.controller;

import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.retry.support.RetryTemplate;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.ArrayList;
import java.util.List;

/**
 * @author cong
 * @date 2021/9/18 18:17
 */
@Slf4j
@RestController
@RequestMapping("retry")
public class RetryTestController {

    @Autowired
    private RetryTemplate template;

    int count = 0;

    @GetMapping("get")
    public List<String> get() {

        ArrayList<String> list = new ArrayList<>();
        template.execute(context -> {
            String s = test();
            list.add(s);
            return StringUtils.isNotBlank(s);
        });
        return list;
    }

    private String test() {
        if (6 > count) {
            log.info(count + "");
            count++;
            throw new RuntimeException("");
        } else {
            log.info(count + "");
            log.info("ok");
            return "ok";
        }

    }
}
```
